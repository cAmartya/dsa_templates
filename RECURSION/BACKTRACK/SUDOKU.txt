#include <iostream>
using namespace std;

bool safe(int **a, int x, int y, int n, int k)
{
    int xlb = x/3, ylb = y/3;
    xlb *= 3; ylb *= 3;
    
    for(int i=xlb;i<=xlb+2;i++)
        for(int j=ylb;j<=ylb+2;j++)
            if(a[i][j] == k)
                return false;
    
	for(int i=0;i<n;i++)
		if(a[i][y]==k)
			return false;
	for(int i=0;i<n;i++)
		if(a[x][i]==k)
			return false;
	
	return true;
}
bool xss(int **a, int x, int n)
{
	if(x>=n)
	{
		return true;
	}
	for(int i=0;i<n;i++)
	{
	    if(a[x][i] > 0){
	        if(xss(a,x+1,n))
			    return true;}
		else
		{
		
	    for(int k=1;k<=n;k++)
	    {
	        if(safe(a, x, i, n, k))
    		{	
	    		a[x][i]=k;
		    	if(xss(a,x+1,n))
			    	return true;
			    a[x][i]=0;
    		}
    	}}
	}
	
	return false;
}

void print(int** a, int n)
{
    for (int row = 0; row < n; row++)
    {
        for (int col = 0; col < n; col++)
            cout << a[row][col] << " ";
        cout << endl;
    }
}
